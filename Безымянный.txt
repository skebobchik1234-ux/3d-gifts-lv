<script>
let cart = [];

function addToCart(name, price){
    cart.push({name, price});
    renderCart();
}

function renderCart(){
    let div = document.getElementById("cartItems");
    div.innerHTML = "";
    cart.forEach(item=>{
        div.innerHTML += `<p>${item.name} - ${item.price}€</p>`;
    });
    updateTotal();
}

function updateTotal(){
    let shipping = parseFloat(document.getElementById("shipping").value);
    let sum = cart.reduce((t,i)=> t+i.price,0);
    document.getElementById("total").innerText = "Итого: " + (sum+shipping) + "€";
}

function checkout(){
    if(cart.length===0){
        alert("Добавьте товар в корзину!");
        return;
    }

    let name = document.getElementById("name").value;
    if(name===""){
        alert("Введите имя!");
        return;
    }

    alert("Спасибо за заказ! Мы свяжемся с вами для подтверждения и оплаты переводом.");
}
